La force brute

Le nombre de combinaisons possibles de 9 sacs de farine est égal à 9!=9×8×7×⋯×2×1=362880. Ce nombre est ridiculement petit pour un ordinateur et la force brute est donc particulièrement bien adaptée au problème.

function solve(row, digit) {
  if (digit < 1) {
    // `row` is a permutation of 1, 2, ..., 9.
    // We have to check if this solution holds.
    var a1 = row[0];
    var b1 = row[1] * 10 + row[2];
    var c = row[3] * 100 + row[4] * 10 + row[5];
    var b2 = row[6] * 10 + row[7];
    var a2 = row[8];
    if (a1 * b1 == c && c == b2 * a2) {
      console.log(a1, b1, c, b2, a2);
    }
    return;
  }
  // Place each digit in all possible places.
  for (var k = 0 ; k < row.length ; k++) {
    if (row[k] < 1) {
      row[k] = digit;
      solve(row, digit - 1);
      row[k] = 0;
    }
  }
}

solve([0,0,0,0,0,0,0,0,0], 9);
